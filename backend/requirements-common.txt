# tier-specific pieces. Split this way because chatterbox-tts and
# kokoro-onnx have directly incompatible numpy requirements (verified: the
# ranges don't overlap at all) — since tier_manager.py already picks
# exactly one TTS engine per install based on detected hardware, no
# environment ever needs both, so the split just matches that reality
# instead of forcing one impossible combined environment.

# SQLCipher 4.x — DB-level encryption of profile.db. The historical
# `sqlcipher3-binary` distribution was discontinued on PyPI, so pin the
# source `sqlcipher3` package directly — since 0.6.0 it ships prebuilt
# wheels for cp39-cp314 on Windows (x86/x64/arm64), macOS
# (universal2/x86_64/arm64), and Linux (x86_64/aarch64/i686 + musllinux),
# so this is a drop-in replacement: import path is unchanged
# (`from sqlcipher3 import dbapi2` in app/db/sqlite_models.py) and PyInstaller
# picks the bundled compiled extension up automatically once the import is
# found (see hearth-backend.spec's hiddenimports and its note about
# collect_dynamic_libs("sqlcipher3") finding nothing).
sqlcipher3==0.6.*

# STT — Moonshine Voice (github.com/moonshine-ai/moonshine), onnx runtime
# backend. The old `useful-moonshine` git package (usefulsensors/moonshine)
# this used to pin no longer installs at all — that repo was rewritten into
# this much larger project and dropped its setup.py/pyproject.toml. Verified
# working end-to-end against 0.0.68 (see app/stt/moonshine_engine.py).
moonshine-voice==0.0.68

# LLM inference server — llama.cpp's `llama-server` binary is NOT installed via pip.
# Build/install separately (https://github.com/ggerganov/llama.cpp) and make sure
# `llama-server` is on PATH, or set LLAMA_SERVER_BIN in config.py (the
# packaged desktop app sets this automatically to a bundled copy — see
# scripts/fetch_llama_cpp.py). Tool calling (used by memory/tools.py) needs
# a recent build with Jinja chat-template support and a model whose
# template llama.cpp maps to tool-call output. The same binary also serves
# EmbeddingGemma-300M (memory/embedder.py), as a second `--embedding`
# process on its own port.
